```shell
#查询索引库中name字段的内容中包含"贾诩"的文档，并使用自定义标签来定义高亮显示的效果
GET myindex_highlight/_doc/_search
{
  "query":{
  "match": {
    "name": "贾诩"
  }
  },
  "highlight":{
    "pre_tags": "<p class='gaoliang'>", #自定义高亮显示效果的标签
    "post_tags": "</p>", 
    "fields": {
      "name": {}
    }
  }
}
```
