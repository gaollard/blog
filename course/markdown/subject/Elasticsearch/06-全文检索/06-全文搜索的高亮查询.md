```shell
#准备数据
POST  /myindex_highlight/_doc
{
  "name":"贾诩",
  "address":"魏国",
  "age":19
}
POST  /myindex_highlight/_doc
{
  "name":"诸葛亮",
  "address":"蜀国",
  "age":37
}

#查询索引库中name字段的内容中包含"贾诩"的文档，并且高亮显示查询的结果
GET myindex_highlight/_doc/_search
{
  "query":{
    "match": {
      "name": "贾诩"
    }
  },
  "highlight":{
    "fields": {
      "name": {}
    }
  }
}
```
